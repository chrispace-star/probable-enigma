    <script>

        document.addEventListener('DOMContentLoaded', function() {

            

            const copyrightYear = document.getElementById('copyright-year');

            if (copyrightYear) {

                copyrightYear.textContent = new Date().getFullYear();

            }


            const form = document.getElementById('roi-form');

            if (form) {

                form.addEventListener('submit', function(event) {

                    event.preventDefault();

                    

                    // --- 1. Get User Inputs ---

                    const monthlyVisitors = parseFloat(document.getElementById('monthly-visitors').value) || 0;

                    const patientValue = parseFloat(document.getElementById('patient-value').value) || 0;

                    const searchUsage = (parseFloat(document.getElementById('search-usage').value) || 0) / 100;

                    const searchFailureRate = (parseFloat(document.getElementById('search-failure-rate').value) || 0) / 100;

                    const patientLeakage = (parseFloat(document.getElementById('patient-leakage').value) || 0) / 100;

                    const callsFromFailure = (parseFloat(document.getElementById('calls-from-failure').value) || 0) / 100;

                    const costPerCall = parseFloat(document.getElementById('cost-per-call').value) || 0;


                    // --- 2. Define Constants ---

                    const ANNUAL_MONTHS = 12;

                    const SEARCHSTAX_IMPROVEMENT_RATE = 0.75; // Assume 75% reduction in search failures


                    // --- 3. Perform Calculations (Before SearchStax) ---

                    const monthlySearchUsers = monthlyVisitors * searchUsage;

                    const monthlyFailedSearches_Before = monthlySearchUsers * searchFailureRate;

                    

                    const annualLostRevenue_Before = monthlyFailedSearches_Before * patientLeakage * patientValue * ANNUAL_MONTHS;

                    const annualCallCosts_Before = monthlyFailedSearches_Before * callsFromFailure * costPerCall * ANNUAL_MONTHS;


                    // --- 4. Perform Calculations (After SearchStax) ---

                    const searchFailureRate_After = searchFailureRate * (1 - SEARCHSTAX_IMPROVEMENT_RATE);

                    const monthlyFailedSearches_After = monthlySearchUsers * searchFailureRate_After;


                    const annualLostRevenue_After = monthlyFailedSearches_After * patientLeakage * patientValue * ANNUAL_MONTHS;

                    const annualCallCosts_After = monthlyFailedSearches_After * callsFromFailure * costPerCall * ANNUAL_MONTHS;


                    // --- 5. Calculate ROI / Savings ---

                    const recoveredRevenue = annualLostRevenue_Before - annualLostRevenue_After;

                    const callCenterSavings = annualCallCosts_Before - annualCallCosts_After;

                    const totalROI = recoveredRevenue + callCenterSavings;


                    // --- 6. Display Results ---

                    displayResults({

                        totalROI,

                        recoveredRevenue,

                        callCenterSavings

                    });

                });

            }


            function displayResults(results) {

                const resultsSection = document.getElementById('results-section');

                

                const currencyFormatter = new Intl.NumberFormat('en-US', { style: 'currency', currency: 'USD', minimumFractionDigits: 0, maximumFractionDigits: 0 });


                document.getElementById('total-roi-value').textContent = currencyFormatter.format(results.totalROI);

                document.getElementById('recovered-revenue').textContent = currencyFormatter.format(results.recoveredRevenue);

                document.getElementById('call-center-savings').textContent = currencyFormatter.format(results.callCenterSavings);


                if (resultsSection.classList.contains('hidden')) {

                    resultsSection.classList.remove('hidden');

                }

                

                resultsSection.scrollIntoView({ behavior: 'smooth' });

            }

        });
